# 3. Create a folder titled your name
mkdir Phil2B

# 4. Create another directory titled biocomputing and change into it
mkdir biocomputing && cd biocomputing

# 5. Download the 3 files
wget https://raw.githubusercontent.com/josoga2/dataset-repos/main/wildtype.fna
wget https://raw.githubusercontent.com/josoga2/dataset-repos/main/wildtype.gbk
wget https://raw.githubusercontent.com/josoga2/dataset-repos/main/wildtype.gbk

# 6. Move the .fna file to the folder titled your name
mv wildtype.fna ../Phil2B/

# 7. Delete duplicate gbk file (wget renames it as .1)
rm -f wildtype.gbk.1

# 8. Confirm if the .fna file is mutant or wild type
grep "tatatata" ../Phil2B/wildtype.fna
grep "tata" ../Phil2B/wildtype.fna

# 9. If mutant, print all matching lines into a new file
grep "tatatata" ../Phil2B/wildtype.fna > mutant_sequences.txt

# 10. Count number of lines (excluding header) in the .gbk file
grep -A1000 ORIGIN wildtype.gbk | grep -v ORIGIN | grep -v "//" | wc -l

# 11. Print the sequence length of the .gbk file
grep "LOCUS" wildtype.gbk

# 12. Print the source organism of the .gbk file
grep "SOURCE" wildtype.gbk

# 13. List all gene names in the .gbk file
grep "/gene=" wildtype.gbk

# 14. Clear your terminal space and print all commands used today
clear
history

# 15. List the files in the two folders
ls   # biocomputing folder
ls ../Phil2B
