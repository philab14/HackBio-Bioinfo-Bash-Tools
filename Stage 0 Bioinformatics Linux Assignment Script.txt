# 1. Print your name
echo " Philip Boakye Bonsu"

# 2. Create a folder titled your name
mkdir Phil2B

# 3. Create another directory titled biocomputing and change into it
mkdir biocomputing && cd biocomputing

# 4. Download the 3 files
wget https://raw.githubusercontent.com/josoga2/dataset-repos/main/wildtype.fna
wget https://raw.githubusercontent.com/josoga2/dataset-repos/main/wildtype.gbk
wget https://raw.githubusercontent.com/josoga2/dataset-repos/main/wildtype.gbk

# 5. Move the .fna file to the folder titled your name
mv wildtype.fna ../Phil2B/

# 6. Delete duplicate gbk file (wget renames it as .1)
rm -f wildtype.gbk.1

# 7. Confirm if the .fna file is mutant or wild type
grep "tatatata" ../Phil2B/wildtype.fna
grep "tata" ../ Phil2B/wildtype.fna

# 8. If mutant, print all matching lines into a new file
grep "tatatata" ../ Phil2B /wildtype.fna > mutant_sequences.txt

# 9. Count number of lines (excluding header) in the .gbk file
sed -n '/^ORIGIN/,/^V/V/p' wildtype.gbk | grep -v 'ORIGIN' | grep -v "//" | wc -L

# 10. Print the sequence length of the .gbk file
grep "Locus" wildtype.gbk

# 11. Print the source organism of the .gbk file
grep "SOURCE" wildtype.gbk

# 12. List all gene names in the .gbk file
grep "/gene=" wildtype.gbk
# 13. Clear your terminal space and print all commands used today
clear
history
# 14. List the files in the two folders
# Files in biocomputing folder
ls
# Files in Phil2B folder
ls ../ Phil2B

conda activate funtools

conda install figlet

# 5. Run figlet <my name>
figlet Philip

conda install -c bioconda bwa

# 7. Install blast through the bioconda channel
conda install -c bioconda blast

# 8. Install samtools through the bioconda channel
conda install -c bioconda samtools

# 9. Install bedtools through the bioconda channel
conda install -c bioconda bedtools

# 10. Install spades.py through the bioconda channel
conda install bioconda::spades

# 11. Install bcftools through the bioconda channel
conda install -c bioconda bcftools

# 12. Install fastp through the bioconda channel
conda install -c bioconda fastp
